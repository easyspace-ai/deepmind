## 修复1：规范化流结束路径

**文件**：`internal/api/langgraph_handler.go`

**问题**：多个分支调用`writeStreamEnd()`，某些错误分支缺少end事件

**改动**：

```go
// ❌ 旧代码（第630-920行）
func (h *LangGraphHandler) streamRun(c *gin.Context) {
    // ...
    
    var endOnce sync.Once
    writeStreamEnd := func(reason string, err error) {
        endOnce.Do(func() {
            endPayload := map[string]interface{}{
                "run_id":    runID,
                "thread_id": threadID,
                "reason":    reason,
            }
            if err != nil {
                endPayload["error"] = err.Error()
            }
            if writeErr := sseWriter.WriteEnd(endPayload); writeErr != nil {
                h.logger.Debug("发送 end 事件失败", zap.Error(writeErr))
            }
        })
    }
    
    // ✅ 改进：增加 defer 作为最终保障
    defer func() {
        // 确保一定发送end事件（防止函数提前返回导致未发end）
        writeStreamEnd("function_exit", nil)
    }()
    
    // ... 检查线程、绑定参数等代码，保持不变
    
    // ⚠️ 关键改动：所有 error return 都要先调用 writeStreamEnd()
    
    if err := sseWriter.WriteMetadata(...); err != nil {
        h.logger.Error("发送 metadata 事件失败", zap.Error(err))
        writeStreamEnd("metadata_error", err)  // ← 改为显式调用
        return  // ← 这样 defer 会再调一遍（但被 sync.Once 挡住）
    }
    
    // ... 用户消息处理等，保持不变
    
    if h.db != nil || h.configLoader != nil {
        // 创建 ChatModelAdapter 部分保持不变...
        
        if chatModel != nil && err == nil {
            // ❌ 旧：模型调用出错时
            if streamErr != nil {
                h.logger.Error("LLM 流式调用失败", zap.Error(streamErr))
                aiResponse.WriteString(fmt.Sprintf("抱歉，LLM 流式调用失败：%v", streamErr))
                // ← 没有 writeStreamEnd()！
            }
            
            // ✅ 改为
            if streamErr != nil {
                h.logger.Error("LLM 流式调用失败", zap.Error(streamErr))
                writeStreamEnd("llm_error", streamErr)  // ← 先发end
                
                // 然后才生成fallback
                aiResponse.WriteString(fmt.Sprintf("抱歉，LLM 流式调用失败：%v", streamErr))
            }
            
            // 流式读取分支
            for {
                chunk, recvErr := streamReader.Recv()
                
                if recvErr != nil {
                    if errors.Is(recvErr, io.EOF) {
                        h.logger.Debug("流式响应读取完毕", zap.Int("totalChunks", chunkCount))
                        // 发送最终消息...保持不变
                    } else {
                        h.logger.Error("流式响应读取错误", zap.Error(recvErr))
                        // ✅ 改：这里也要writeStreamEnd()
                        writeStreamEnd("stream_recv_error", recvErr)
                    }
                    break  // ← 此后会继续执行下面的 values/end
                }
                
                // ... chunk 处理逻辑保持不变
            }
        }
    }
    
    stopKeepAliveFn()
    
    // Fallback 和 values 处理保持不变...
    
    // 最后显式调用一次，执行最后的end事件
    writeStreamEnd("completed", nil)
}
```

## 修复2：改进客户端断开检测

**文件**：`internal/api/langgraph_handler.go`

**问题**：`streamReader.Recv()` 是阻塞的，无法及时响应客户端断开

**改动**：

```go
// 在读取chunk的循环中增加周期检查

// ❌ 旧代码
for {
    // 这个 Recv() 可能阻塞 5-30 秒都不返回
    chunk, recvErr := streamReader.Recv()
    chunkCount++
    
    if recvErr != nil {
        // 直到出错才能检查中断
        // ...
    }
}

// ✅ 改为：定时检查客户端状态
for {
    // 方案1：使用 select + 超时，避免长时间阻塞
    recvDone := make(chan interface{}, 1)
    go func() {
        chunk, recvErr := streamReader.Recv()
        recvDone <- map[string]interface{}{
            "chunk":   chunk,
            "err":     recvErr,
        }
    }()
    
    select {
    case <-c.Request.Context().Done():
        // 客户端断开，立即返回
        h.logger.Debug("客户端断开连接")
        writeStreamEnd("client_disconnected", nil)
        stopKeepAliveFn()
        return
        
    case <-time.After(8 * time.Second):  // 8s超过keep-alive间隔(5s)，给点余量
        // 如果8s还没有新chunk...
        h.logger.Debug("等待chunk超时，可能网络卡住或客户端已断开，继续keep-alive")
        // 继续loop，这样会再次检查 Context.Done()
        continue
        
    case result := <-recvDone:
        if result != nil {
            res := result.(map[string]interface{})
            chunk := res["chunk"].(*schema.Message)
            recvErr := res["err"].(error)
            chunkCount++
            
            if recvErr != nil {
                // 原有的错误处理...
                if errors.Is(recvErr, io.EOF) {
                    h.logger.Debug("流式响应读取完毕")
                    // ...
                } else {
                    h.logger.Error("流式响应读取错误", zap.Error(recvErr))
                    writeStreamEnd("stream_recv_error", recvErr)
                }
                break
            }
            
            // chunk 处理逻辑...保持不变
        }
    }
}

// 方案2（更简单但要求 streamReader 支持 context）
// 需要改造 schema.StreamReader 为支持 context 的版本
// 这涉及改动Eino框架，暂时不做，先用方案1
```

**风险提示**：

- 新建goroutine会有轻微的性能开销（但可以接受）
- 需要验证`recvDone`不会导致goroutine泄漏（应该不会，因为select只要有结果就返回）

## 修复3：增加详细的错误日志

**文件**：`internal/api/langgraph_handler.go`

**问题**：缺少详细的时间戳和事件追踪，难以排查

**改动**：

```go
func (h *LangGraphHandler) streamRun(c *gin.Context) {
    threadID := c.Param("threadId")
    runID := uuid.NewString()
    
    // ✅ 增加起始时间戳（所有日志都用相对时间）
    startTime := time.Now()
    logTiming := func(stage string, details ...interface{}) {
        elapsed := time.Since(startTime)
        h.logger.Info("SSE流程追踪",
            zap.String("run_id", runID),
            zap.String("stage", stage),
            zap.Duration("elapsed_ms", elapsed),
            zap.Any("details", details))
    }
    
    logTiming("start")
    
    // ... 检查线程、绑定参数
    
    sseWriter := deerflow.NewWriter(runID, c.Writer)
    logTiming("sseWriter_created")
    
    if err := sseWriter.WriteMetadata(...); err != nil {
        logTiming("metadata_error", err)
        return
    }
    logTiming("metadata_sent")
    
    // ... 用户消息处理
    
    if err := sseWriter.WriteMessagesTuple(...); err != nil {
        logTiming("user_message_error", err)
        return
    }
    logTiming("user_message_sent")
    
    // keep-alive goroutine
    keepAliveCount := 0
    go func() {
        for {
            select {
            case <-keepAliveTicker.C:
                keepAliveCount++
                if err := sseWriter.WriteKeepAlive(); err != nil {
                    h.logger.Debug("keep-alive发送失败", 
                        zap.Int("count", keepAliveCount),
                        zap.Error(err))
                    return
                }
                logTiming("keep_alive_sent", keepAliveCount)  // 可选，降低日志量
            case <-stopKeepAlive:
                return
            case <-c.Request.Context().Done():
                return
            }
        }
    }()
    
    // ... LLM调用
    
    logTiming("llm_call_start")
    streamReader, streamErr := chatModel.Stream(streamCtx, einoMessages)
    if streamErr != nil {
        logTiming("llm_call_error", streamErr)
        // ...
    }
    logTiming("llm_stream_created")
    
    // 流读取循环
    chunkCount := 0
    for {
        chunk, recvErr := streamReader.Recv()
        chunkCount++
        
        if recvErr != nil {
            logTiming("stream_recv_error", map[string]interface{}{
                "total_chunks": chunkCount,
                "error":        recvErr.Error(),
                "is_eof":       errors.Is(recvErr, io.EOF),
            })
            // ...
        } else {
            logTiming("stream_recv_ok", map[string]interface{}{
                "chunk_size": len(chunk.Content),
            })
        }
    }
    
    logTiming("values_start")
    if err := sseWriter.WriteValues(...); err != nil {
        logTiming("values_error", err)
        return
    }
    logTiming("values_sent")
    
    writeStreamEnd("completed", nil)
    logTiming("end_sent")
    
    logTiming("function_exit")
}
```

## 修复4：改进 WriteKeepAlive() 的并发安全性

**文件**：`pkg/sse/writer.go`

**问题**：`WriteKeepAlive()` 调用 `WriteComment()`，而后者已有互斥保护，理论上安全，但可以更明确

**改动**（可选，留作文档）：

```go
// 原代码已经是对的，这里只是加如下注释确保意图清晰

func (w *Writer) WriteKeepAlive() error {
    // 注意：WriteComment() 内部已有 w.mu 互斥保护
    // 即使从多个goroutine并发调用 WriteKeepAlive()，
    // 也能保证对 http.ResponseWriter 的写入是串行的
    return w.WriteComment("keep-alive")
}

// 更安全的替代方案（可选）：
func (w *Writer) WriteKeepAliveUnsafe() error {
    // 这个版本假设调用者已经持有锁，速度更快
    // 但当前代码不需要这个优化
    if w.closed {
        return nil
    }
    _, err := fmt.Fprintf(w.w, ": keep-alive\n\n")
    if err != nil {
        w.closed = true
        return err
    }
    w.flusher.Flush()
    return nil
}
```

    t.Logf("Keep-alive events: %d in %.1fs", keepAliveCount, elapsed.Seconds())

func TestStreamRunClientDisconnect(t *testing.T) {
    handler := NewTestHandler()
    
    // 创建一个可以取消的context
    ctx, cancel := context.WithCancel(context.Background())
    
    req := httptest.NewRequest(
        "POST",
        "/api/langgraph/threads/test-789/runs/stream",
        strings.NewReader(`{"assistant_id":"lead_agent","messages":[]}`),
    ).WithContext(ctx)
    req.Header.Set("Content-Type", "application/json")
    
    w := httptest.NewRecorder()
    
    var done sync.WaitGroup
    done.Add(1)
    
    go func() {
        defer done.Done()
        handler.streamRun(w.Header(), w)
    }()
    
    // 等待一些时间让请求启动
    time.Sleep(500 * time.Millisecond)
    
    // 模拟客户端断开
    cancel()
    
    // 应该在1秒内返回（不会无限等待）
    doneChan := make(chan struct{})
    go func() {
        done.Wait()
        doneChan <- struct{}{}
    }()
    
    select {
    case <-doneChan:
        t.Log("Handler returned promptly after client disconnect")
    case <-time.After(3 * time.Second):
        t.Fatal("Handler did not return within 3 seconds after client disconnect")
    }
    
    body := w.Body.String()
    assert.Contains(t, body, "client_disconnected", "应该包含client_disconnected原因")
}
