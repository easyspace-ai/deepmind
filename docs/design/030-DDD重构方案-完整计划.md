### 1.1 Agent 编排上下文 (Agent Orchestration)

**定义**: 管理 AI Agent 的身份、能力、配置、执行状态。

#### 聚合根
```
Aggregate Root: Agent
  - AgentID (ID)
  - Identity (name, description, role)
  - Capabilities (tools, models, skills)
  - PersonalityConfig (thinking_style, response_style)
  - Metadata (created_at, version)
```

#### 关键实体
- `AgentIdentity` - Agent 的核心身份（不可变值对象）
- `AgentCapabilities` - Agent 的能力集合
- `PromptTemplate` - 提示词模板（包含提示词部分：Role、Thinking、Skills 等）

#### 聚合内的业务规则
```go
// 示例规则
func (a *Agent) AddCapability(capability Capability) error {

    a.RaiseDomainEvent(CapabilityAddedEvent{})
    return nil
}

func (a *Agent) BuildSystemPrompt(context RequestContext) string {
    // 模块化提示词构建（参考 DeerFlow）

        a.Identity,
        a.capabilities,
        context.CurrentMemory,
        context.ActiveSkills,
```

### 1.2 会话上下文 (Session Context)

**定义**: 管理用户与 Agent 的对话生命周期。

#### 聚合根
```
Aggregate Root: ConversationSession
  - SessionID (ID)
  - UserCode (租户标识)
  - AgentID
  - State (active, archived, failed)
  - ThreadData (虚拟路径、工作目录、上传文件)
  - Metadata
```

#### 关键实体
- `Message` - 不可变消息记录（包含 role、content、toolCalls）
- `ThreadData` - 会话的虚拟文件系统状态
- `ToolCallState` - 工具调用的挂起/完成状态

#### 业务规则
```go
// 示例规则
func (s *ConversationSession) AppendMessage(msg Message) error {
    if s.State != StateActive {
        return ErrSessionNotActive
    }
    
    if !s.canProcessMessage(msg) {
        return ErrInvalidMessageState
    }
    
    s.messages = append(s.messages, msg)
    s.RaiseDomainEvent(MessageAppendedEvent{Message: msg})
    return nil
}

// 处理悬挂的工具调用
func (s *ConversationSession) ResolvePendingToolCall(
    toolCallID string, result ToolCallResult,
) error {
    state := s.threadData.GetToolCallState(toolCallID)
    if state == nil {
        return ErrToolCallNotFound
    }
    
    state.Result = result
    state.Status = Completed
    s.RaiseDomainEvent(ToolCallResolvedEvent{})
    return nil
}
```

```go
    tools map[string]Tool // name -> Tool
}

func (tr *ToolRegistry) RegisterTool(server MCPServer, tool Tool) error {
    fullName := fmt.Sprintf("%s:%s", server.Name(), tool.Name)
    if _, exists := tr.tools[fullName]; exists {
        return ErrToolAlreadyExists
    }
    tr.tools[fullName] = tool
    tr.RaiseDomainEvent(ToolRegisteredEvent{})
    return nil
}
```

```go
// 具体中间件示例
type ThreadDataMiddleware struct {
    sessionRepo SessionRepository
}

func (m *ThreadDataMiddleware) Execute(
    ctx context.Context, mctx *MiddlewareContext,
) error {
    // 为会话创建虚拟线程目录
    session, err := m.sessionRepo.FindByID(ctx, mctx.SessionID)
    if err != nil {
        return err
    }
    
    if session.ThreadData == nil {
        session.ThreadData = NewThreadData(mctx.UserCode, mctx.SessionID)
        if err := m.sessionRepo.Save(ctx, session); err != nil {
            return err
        }
    }
    
    mctx.ThreadData = session.ThreadData
    return nil
}
```

## Part 3: Agent 系统重构 (Phase 3)

### 3.1 Agent 执行引擎重设计

```go
// 旧架构问题：
// - Agent 执行逻辑散落在多处
// - 工具调用与领域规则未分离
// - 提示词生成过于简单

// 新架构：
type AgentExecutor interface {
    Execute(ctx context.Context, req ExecuteRequest) (Response, error)
}

type ExecuteRequest struct {
    SessionID     string
    UserMessage   string
    SystemPrompt  string
    Tools         []Tool
    History       []Message
    Model         LLMProvider
}

type Response struct {
    SessionID   string
    MessageID   string
    Content     Content // 类型化的响应内容
    ToolCalls   []PendingToolCall
    Metadata    ResponseMetadata
}

// 具体实现
type EinoAgentExecutor struct {
    eino          *eino.Agent
    eventBus      EventBus
    sessionRepo   SessionRepository
    toolRegistry  ToolRegistry
}

func (e *EinoAgentExecutor) Execute(
    ctx context.Context, req ExecuteRequest,
) (Response, error) {
    // 1. 应用中间件链
    chain := e.createMiddlewareChain()
    mctx := &MiddlewareContext{/*...*/}
    if err := chain.Execute(ctx, mctx); err != nil {
        return Response{}, err
    }
    
    // 2. 构建提示词
    systemPrompt := e.buildSystemPrompt(ctx, req)
    
    // 3. 调用 LLM
    aiResponse, err := e.eino.Invoke(ctx, &EinoRequest{
        SystemPrompt: systemPrompt,
        Messages:     req.History,
        Tools:        req.Tools,
    })
    if err != nil {
        return Response{}, err
    }
    
    // 4. 解析响应
    response := e.parseResponse(aiResponse)
    
    // 5. 发布事件
    e.eventBus.Publish(ctx, &MessageAppendedEvent{
        SessionID: req.SessionID,
        Message:   response.Message,
    })
    
    return response, nil
}
```

### 3.3 沙箱系统（参考 DeerFlow）

```go
// 虚拟路径系统
type VirtualPath string // e.g., /mnt/user-data/workspace/file.txt

type PhysicalPath string // e.g., /tmp/user-123/workspace/file.txt

// Sandbox 接口
type Sandbox interface {
    // 虚拟 -> 物理路径
    TranslateVirtualPath(vpath VirtualPath) (PhysicalPath, error)
    
    // 物理 -> 虚拟路径（用于错误消息）
    TranslatePhysicalPath(ppath PhysicalPath) (VirtualPath, error)
    
    // 路径验证
    ValidatePath(vpath VirtualPath) error
    
    // 执行带有沙箱限制的操作
    ExecuteInSandbox(ctx context.Context, op Operation) (OperationResult, error)
}

// 操作类型
type Operation interface {
    Execute(ctx context.Context, rootPath PhysicalPath) error
}

// 具体实现
type FilesystemSandbox struct {
    userCode       string
    sessionID      string
    baseDir        string // /tmp/nanobot
    workspaceDir   string // /tmp/nanobot/user-123/workspace
    uploadsDir     string // /tmp/nanobot/user-123/uploads
    outputsDir     string // /tmp/nanobot/user-123/outputs
    skillsDir      string // /tmp/nanobot/skills (readonly)
}

func (s *FilesystemSandbox) TranslateVirtualPath(vpath VirtualPath) (PhysicalPath, error) {
    vstr := string(vpath)
    
    // /mnt/user-data/workspace/* -> {workspaceDir}/*
    if strings.HasPrefix(vstr, "/mnt/user-data/workspace/") {
        rel := strings.TrimPrefix(vstr, "/mnt/user-data/workspace/")
        return PhysicalPath(filepath.Join(s.workspaceDir, rel)), nil
    }
    
    // /mnt/user-data/uploads/* -> {uploadsDir}/*
    if strings.HasPrefix(vstr, "/mnt/user-data/uploads/") {
        rel := strings.TrimPrefix(vstr, "/mnt/user-data/uploads/")
        return PhysicalPath(filepath.Join(s.uploadsDir, rel)), nil
    }
    
    // /mnt/user-data/outputs/* -> {outputsDir}/*
    if strings.HasPrefix(vstr, "/mnt/user-data/outputs/") {
        rel := strings.TrimPrefix(vstr, "/mnt/user-data/outputs/")
        return PhysicalPath(filepath.Join(s.outputsDir, rel)), nil
    }
    
    // /mnt/skills/* -> {skillsDir}/* (readonly)
    if strings.HasPrefix(vstr, "/mnt/skills/") {
        rel := strings.TrimPrefix(vstr, "/mnt/skills/")
        return PhysicalPath(filepath.Join(s.skillsDir, rel)), nil
    }
    
    return "", ErrInvalidVirtualPath
}

func (s *FilesystemSandbox) ValidatePath(vpath VirtualPath) error {
    ppath, err := s.TranslateVirtualPath(vpath)
    if err != nil {
        return err
    }
    
    // 防止路径遍历
    realPath, err := filepath.EvalSymlinks(string(ppath))
    if err != nil {
        return ErrPathEvaluationFailed
    }
    
    // 检查是否在允许的基础目录内
    for _, allowed := range []string{s.workspaceDir, s.uploadsDir, s.outputsDir, s.skillsDir} {
        if strings.HasPrefix(realPath, allowed) {
            return nil
        }
    }
    
    return ErrPathTraversalDetected
}
```

## Part 4: 消息处理流程重构 (Phase 4)

### 4.1 完整消息处理流程

```
用户消息 (来自单一或多渠道)
    ↓
[消息适配器] - 转换为领域消息
    ↓
[鉴权中间件] - 验证用户
    ↓
[消息中间件链] - ThreadData、Uploads、Summarization、Memory 等
    ↓
[会话恢复] - 加载或创建 ConversationSession
    ↓
[Agent 执行器]
    ├─ 构建系统提示词          (提示词构建器)
    ├─ 加载可用工具             (工具注册表)
    ├─ 调用 LLM                (Eino Agent)
    └─ 解析响应                 (响应解析器)
    ↓
[工具调用执行]
    ├─ 验证工具                 (工具验证)
    ├─ 沙箱执行                 (Sandbox)
    └─ 收集结果                 (结果聚合)
    ↓
[事件发布] - MessageAppendedEvent、ToolCallResolvedEvent
    ↓
[结果落盘] - 保存到数据库
    ↓
[渠道适配器] - 转换为渠道消息
    ↓
[多渠道分发] - 发送给用户
```

### 4.2 消息处理服务 (Domain Service)

```go
type MessageProcessingService struct {
    middlewareChain    MiddlewareChain
    sessionRepo        SessionRepository
    agentRepo          AgentRepository
    agentExecutor      AgentExecutor
    toolRegistry       ToolRegistry
    sandbox            Sandbox
    eventBus           EventBus
    channelAdapters    map[string]ChannelAdapter
}

func (s *MessageProcessingService) ProcessMessage(
    ctx context.Context,
    incoming IncomingMessage,
) error {
    // 1. 认证
    user, err := s.authenticate(ctx, incoming.UserCode)
    if err != nil {
        return err
    }
    
    // 2. 应用中间件链
    mctx := &MiddlewareContext{
        SessionID: incoming.SessionID,
        UserCode:  user.UserCode,
    }
    if err := s.middlewareChain.Execute(ctx, mctx); err != nil {
        s.publishError(ctx, incoming.SessionID, err)
        return nil // 错误已处理，不需要返回
    }
    
    // 3. 恢复或创建会话
    session, err := s.getOrCreateSession(ctx, incoming)
    if err != nil {
        return err
    }
    
    // 4. 追加用户消息到会话
    userMessage := Message{
        ID:    uuid.New().String(),
        Role:  RoleUser,
        Content: &TextContent{Text: incoming.Text},
    }
    if err := session.AppendMessage(userMessage); err != nil {
        return err
    }
    
    // 5. 执行 Agent
    response, err := s.agentExecutor.Execute(ctx, ExecuteRequest{
        SessionID:    session.ID,
        UserMessage:  incoming.Text,
        Tools:        s.toolRegistry.GetToolsForAgent(session.AgentID),
        History:      session.Messages,
        Model:        session.LLMProvider,
    })
    if err != nil {
        return err
    }
    
    // 6. 处理工具调用
    for _, toolCall := range response.ToolCalls {
        result, err := s.executeToolCall(ctx, toolCall)
        if err != nil {
            // 错误结果也要追加
            result = ToolResultContent{
                ToolCallID: toolCall.ID,
                IsError:    true,
                Result:     err.Error(),
            }
        }
        
        toolResultMsg := Message{
            ID:      uuid.New().String(),
            Role:    RoleToolResult,
            Content: &result,
        }
        session.AppendMessage(toolResultMsg)
    }
    
    // 7. 保存会话
    if err := s.sessionRepo.Save(ctx, session); err != nil {
        return err
    }
    
    // 8. 发送给用户
    domainMessage := response.Message
    for _, adapter := range s.channelAdapters {
        channelMessage := adapter.FromDomain(domainMessage)
        if err := adapter.Send(ctx, channelMessage); err != nil {
            logger.Error("failed to send message", zap.Error(err))
            // 继续尝试其他渠道
        }
    }
    
    return nil
}

func (s *MessageProcessingService) executeToolCall(
    ctx context.Context,
    toolCall PendingToolCall,
) (interface{}, error) {
    tool, err := s.toolRegistry.GetTool(toolCall.Name)
    if err != nil {
        return nil, err
    }
    
    // 在沙箱中执行
    result, err := s.sandbox.ExecuteInSandbox(ctx, &ToolCallOperation{
        Tool: tool,
        Args: toolCall.Arguments,
    })
    
    return result, err
}
```

```go
// 基础仓储接口
type Repository[T any] interface {
    FindByID(ctx context.Context, id ID) (*T, error)
    Save(ctx context.Context, entity *T) error
    Delete(ctx context.Context, id ID) error
    List(ctx context.Context, spec Specification[T]) ([]*T, error)
}

// 具体仓储
type SessionRepository interface {
    Repository[ConversationSession]
    FindByUserCode(ctx context.Context, userCode string) ([]*ConversationSession, error)
    FindActive(ctx context.Context, userCode string) ([]*ConversationSession, error)
}

type AgentRepository interface {
    Repository[Agent]
    FindByUserCode(ctx context.Context, userCode string) ([]*Agent, error)
}

// 仓储工厂
type RepositoryFactory interface {
    CreateSessionRepository() SessionRepository
    CreateAgentRepository() AgentRepository
    CreateUserRepository() UserRepository
    // ...
}

// GORM 实现示例
type GormSessionRepository struct {
    db *gorm.DB
}

func (r *GormSessionRepository) FindByID(
    ctx context.Context, id ID,
) (*ConversationSession, error) {
    var session ConversationSession
    if err := r.db.WithContext(ctx).First(&session, "id = ?", id).Error; err != nil {
        return nil, err
    }
    return &session, nil
}

func (r *GormSessionRepository) Save(
    ctx context.Context, session *ConversationSession,
) error {
    return r.db.WithContext(ctx).Save(session).Error
}
```
